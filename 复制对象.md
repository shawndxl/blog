var a1 = a;

var b = new Object(a);

var c = Object.create(a);

var d = Object.assign(a);

var e = JSON.parse(JSON.stringify(a));

var f =  {};
Object.keys(a).forEach((i) => f[i] = a [i]);

var g = $.extend({}, a);

// 查看初始值
console.log(a)
console.log(b)
console.log(c); console.log(c.s);
console.log(d)
console.log(e)
console.log(f)
